\section{Himpunan penyelesaian dari sistem persamaan 2x+y=4 dan 3x+y=6}
\begin{align*}
  3x+y=6\\
  y=6-3x\\
  2x+2(6-3x)=4\\
  2x+12-6x=4\\
  -4x=4-12\\
  -4x=-8\\
  x=2\\
\end{align*}
\begin{align*}
  y=6-3(2)
  y=6-6
  y=0
\end{align*}
Himpunan penyelesaiannya adalah 
\begin{align*}
  x=2
  y=0
\end{align*}
