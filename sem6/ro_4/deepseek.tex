\section*{Soal 1: Masalah Optimasi Produksi Mebel}

\begin{align*}
\text{Maksimumkan } & Z = 250000x_1 + 400000x_2 \\
\text{Dengan batasan: } & \\
& x_1 + 3x_2 \leq 18 \\
& 2x_1 + 2x_2 \leq 24 \\
& x_1, x_2 \geq 0
\end{align*}
